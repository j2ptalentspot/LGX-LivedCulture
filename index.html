⚠️ Database table not set up yet

Run this SQL in your Supabase Dashboard → SQL Editor to create the practitioners table:

CREATE TABLE IF NOT EXISTS practitioners (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  created_at timestamptz DEFAULT now(),
  full_name text, email text UNIQUE, mobile text,
  city text, country text, linkedin_url text, website_url text,
  role text, organization text, years_exp int,
  practitioner_type text[], competency_domains text[],
  frameworks_used text[], industries_served text[],
  contribution_interests text[], engagement_types text[],
  geography text[], bio text, research_interests text,
  services_offered text, fee_range text, availability text,
  signature_offering text, intro_video_url text,
  papers jsonb, videos jsonb, referral_source text,
  is_verified boolean DEFAULT false, status text DEFAULT 'pending'
);
ALTER TABLE practitioners ENABLE ROW LEVEL SECURITY;
CREATE POLICY "anon_insert" ON practitioners FOR INSERT TO anon WITH CHECK (true);
CREATE POLICY "anon_read_own" ON practitioners FOR SELECT TO anon USING (true);
